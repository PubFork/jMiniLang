import "sys.base";
var g_range_array = yield ~(a) {
    var size = g_array_size(a);
    for (var i = 0; i < size; i++) {
        yield g_array_get(a, i);
    }
};
export "g_range_array";
var g_array_find = func ~(a, b) {
    var size = g_array_size(a);
    for (var i = 0; i < size; i++) {
        if (b == g_array_get(a, i)) {
            return i;
        }
    }
};
export "g_array_find";
